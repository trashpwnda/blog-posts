### Fenced Code Block

```js
// Function with default parameters and template literals
function greet(name = "World", enthusiastic = true) {
  let message = `Hello, ${name}`;
  if (enthusiastic) {
    message += "!!!";
  }
  return message;
}

// Array, object, loop, and map
const names = ["Alice", "Bob", "Charlie"];
const greetings = {};
names.forEach((name) => {
  greetings[name] = greet(name);
});

// Class with constructor, method, and static method
class Circle {
  constructor(radius) {
    this.radius = radius;
  }

  area() {
    return Math.PI * this.radius ** 2;
  }

  static describe() {
    return "A shape with all points equidistant from the center.";
  }
}

// Try-catch block
try {
  const c = new Circle(5);
  console.log(`Area: ${c.area().toFixed(2)}`);
  console.log(Circle.describe());
  console.log(greetings["Daisy"]); // undefined
} catch (error) {
  console.error("Something went wrong:", error);
}

// Arrow function and filter
const evenNumbers = [...Array(10).keys()].filter((x) => x % 2 === 0);
console.log("Even numbers:", evenNumbers);
```

### Indented Code Block

```py
import math

# Function with default and keyword arguments
def greet(name="World", enthusiastic=True):
    message = f"Hello, {name}"
    if enthusiastic:
        message += "!!!"
    return message

# List, dictionary, loop, and list comprehension
names = ["Alice", "Bob", "Charlie"]
greetings = {name: greet(name) for name in names}

# Class with constructor, instance method, and static method
class Circle:
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        return math.pi * self.radius ** 2

    @staticmethod
    def describe():
        return "A shape with all points equidistant from the center."

# Exception handling
try:
    c = Circle(5)
    print(f"Area: {c.area():.2f}")
    print(Circle.describe())
    print(greetings["Daisy"])  # KeyError
except KeyError as e:
    print(f"Key error: {e}")

# Lambda and filter
even_numbers = list(filter(lambda x: x % 2 == 0, range(10)))
print("Even numbers:", even_numbers)
```
